    }

        // Ensure the provided reward amount is not more than the balance in the contract.

        // Reward + leftover must be less than 2^256 / 10^18 to avoid overflow.

        );
 
        lastUpdateTime = block.timestamp;

        emit RewardAdded(reward);
    }
